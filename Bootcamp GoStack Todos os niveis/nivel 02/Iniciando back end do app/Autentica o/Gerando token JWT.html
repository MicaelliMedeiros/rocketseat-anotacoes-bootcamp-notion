Importar o jsonwebtoken

AuthenticateUserService.ts

import { getRepository } from 'typeorm';
import { compare } from 'bcryptjs';
import { sign } from 'jsonwebtoken';

import User from '../models/User';

interface Request {
  email: string;
  password: string;
}

interface Response {
  user: User;
  token: string;
}

class AuthenticateUserService {
  public async execute({ email, password }: Request): Promise<Response> {
    const usersRepository = getRepository(User);

    // Verificar se existe o email na base de dados.
    const user = await usersRepository.findOne({
      where: { email },
    });

    // Se nao existir o email, dar ERRO.
    if (!user) {
      throw new Error('Incorrect email/password combination');
    }

    // user.password - Senha criptografada
    // password - Senha nao criptografada

    // Vai comparar se a senha digitada e criptografada sao iguais.
    const passwordMatched = await compare(password, user.password);

    if (!passwordMatched) {
      throw new Error('Incorrect email/password combination');
    }

    // Fazendo um token
    // sign({ Payload }, 'Chave Secreta', { Configuracoes do Token })
    // sign {} Payload - Informacoes do usuario para utilizar depois. Ex.: Permissoes, nome do usuario. Nao pode informacoes segurar como email ou senha.
    // sign '' Chave secreata - md5 online - 54f15c68e0c65c99a450c519379e4b73
    // sign {} - Configuracoes do token

    const token = sign({}, '54f15c68e0c65c99a450c519379e4b73', {
      subject: user.id, // Id do usuario que gerou esse token
      expiresIn: '1d', // Quanto tempo o token vai durar
    });

    return {
      user,
      token,
    };
  }
}

export default AuthenticateUserService;

Buscar o token do usuario no execute, e retornar ele no return

sessions.routes.ts

import { Router } from 'express';

import AuthenticateUserService from '../services/AuthenticateUserService';

const sessionsRouter = Router();

sessionsRouter.post('/', async (request, response) => {
  try {
    const { email, password } = request.body;

    // Criar um service para fazer as regras de negocio
    const authenticateUser = new AuthenticateUserService();

    const { user, token } = await authenticateUser.execute({
      email,
      password,
    });

    // Nao mostrar a senha no return de user.
    delete user.password;

    return response.json({ user, token });
  } catch (err) {
    return response.status(400).json({ error: err.message });
  }
});

export default sessionsRouter;
