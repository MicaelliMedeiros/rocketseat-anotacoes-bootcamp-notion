AuthenticateUserService.ts

import { getRepository } from 'typeorm';
import { compare } from 'bcryptjs';
import { sign } from 'jsonwebtoken';
import authConfig from '../config/auth';

import AppError from '../errors/AppError';

import User from '../models/User';

interface Request {
  email: string;
  password: string;
}

interface Response {
  user: User;
  token: string;
}

class AuthenticateUserService {
  public async execute({ email, password }: Request): Promise<Response> {
    const usersRepository = getRepository(User);

    // Verificar se existe o email na base de dados.
    const user = await usersRepository.findOne({
      where: { email },
    });

    // Se nao existir o email, dar ERRO.
    if (!user) {
      throw new AppError('Incorrect email/password combination', 401);
    }

    // user.password - Senha criptografada
    // password - Senha nao criptografada

    // Vai comparar se a senha digitada e criptografada sao iguais.
    const passwordMatched = await compare(password, user.password);

    if (!passwordMatched) {
      throw new AppError('Incorrect email/password combination', 401);
    }

    // Fazendo um token
    // sign({ Payload }, 'Chave Secreta', { Configuracoes do Token })
    // sign {} Payload - Informacoes do usuario para utilizar depois. Ex.: Permissoes, nome do usuario. Nao pode informacoes segurar como email ou senha.
    // sign '' Chave secreata - md5 online - 54f15c68e0c65c99a450c519379e4b73
    // sign {} - Configuracoes do token

    const { secret, expiresIn } = authConfig.jwt;

    const token = sign({}, secret, {
      subject: user.id, // Id do usuario que gerou esse token
      expiresIn, // Quanto tempo o token vai durar
    });

    return {
      user,
      token,
    };
  }
}

export default AuthenticateUserService;

CreateAppointmentService.ts

import { startOfHour } from 'date-fns';
import { getCustomRepository } from 'typeorm';

import AppError from '../errors/AppError';

import Appointment from '../models/Appointment';
import AppointmentsRepository from '../repositories/AppointmentsRepository';

interface Request {
  provider_id: string;
  date: Date;
}

class CreateAppointmentService {
  public async execute({ date, provider_id }: Request): Promise<Appointment> {
    const appointmentsRepository = getCustomRepository(AppointmentsRepository);

    const appointmentDate = startOfHour(date);

    // Verificar se existe uma data ja agendada.
    const findAppointmentInSameDate = await appointmentsRepository.findByDate(
      appointmentDate,
    );

    if (findAppointmentInSameDate) {
      throw new AppError('This appointment is already booked');
    }

    const appointment = appointmentsRepository.create({
      provider_id,
      date: appointmentDate,
    });

    await appointmentsRepository.save(appointment);

    return appointment;
  }
}

export default CreateAppointmentService;

CreateUserService.ts

import { getRepository } from 'typeorm';
import { hash } from 'bcryptjs';

import AppError from '../errors/AppError';

import User from '../models/User';

interface Request {
  name: string;
  email: string;
  password: string;
}

// Se for usar os metodos padroes do TypeORM(create,update), nao precisar criar um metodo adicional(findByDate), nao preciso de repositorio, apenas usar o getRepository do TypeORM
class CreateUserService {
  public async execute({ name, email, password }: Request): Promise<User> {
    const usersRepository = getRepository(User);

    // Verifica se existe um usuario com um email ja criado
    const checkUserExists = await usersRepository.findOne({
      where: { email },
    });

    // Se existir, dar ERRO
    if (checkUserExists) {
      throw new AppError('Email address already used.');
    }

    // Cria a hash no password
    const hashedPassword = await hash(password, 8);

    // Criar uma instacia para salvar no banco
    const user = usersRepository.create({
      name,
      email,
      password: hashedPassword,
    });

    // Salvando o usuario no banco de dados
    await usersRepository.save(user);

    return user;
  }
}

export default CreateUserService;

UpdateUserAvatarService.ts

import { getRepository } from 'typeorm';
import path from 'path';
import fs from 'fs';
import uploadConfig from '../config/upload';

import AppError from '../errors/AppError';

import User from '../models/User';

interface Request {
  user_id: string;
  avatarFilename: string;
}

class UpdateUserAvatarService {
  public async execute({ user_id, avatarFilename }: Request): Promise<User> {
    const usersRepository = getRepository(User);

    // Verificar ser o id eh de um id de um usuario valido
    const user = await usersRepository.findOne(user_id); // Aqui ele vai procurar diretamente pelo id

    if (!user) {
      throw new AppError('Only authenticated users can change avatar.', 401);
    }

    // Se o usuario ja tiver um avatar
    if (user.avatar) {
      // Deletar avatar anterior. Vai buscar o arquivo de avatar do usuario
      // Pegar a directory, que vai ser importada da uploadConfig
      // Unir dois caminhos, primeiro caminho = uploadConfig.directory, depois o arquivo para remover = user.avatar
      const userAvatarFilePath = path.join(uploadConfig.directory, user.avatar);

      // Verificar se o arquivo realmente existe. Stat traz o status de um arquivo somente se ele existir.
      const userAvatarFileExists = await fs.promises.stat(userAvatarFilePath);

      // Se o arquivo existir, vai deletar
      if (userAvatarFileExists) {
        await fs.promises.unlink(userAvatarFilePath);
      }
    }

    // Vai salvar no user.avatar o avatarFilename, que eh passado por parametro
    user.avatar = avatarFilename;

    // Vai salvar no banco de dados a informacao do user, a informacao do avatar
    await usersRepository.save(user);

    return user;
  }
}

export default UpdateUserAvatarService;

ensureAuthenticated.ts

import { Request, Response, NextFunction } from 'express';
import { verify } from 'jsonwebtoken';

import authConfig from '../config/auth';

import AppError from '../errors/AppError';

interface TokenPayload {
  iat: number;
  exp: number;
  sub: string;
}

// Se for validado/autenticado vai chamar o next
export default function ensureAuthenticated(
  request: Request,
  response: Response,
  next: NextFunction,
): void {
  // Validacao do token JWT
  const authHeader = request.headers.authorization;

  if (!authHeader) {
    throw new AppError('JWT token is missing', 401);
  }

  // Bearer aodkaowjTOKEN
  // Vai dividir pelo espaco
  const [, token] = authHeader.split(' ');

  try {
    // token, e o segundo parametro vai ser o secret. Vou abstrair em um arquivo na pasta config no arquivo auth.ts
    const decoded = verify(token, authConfig.jwt.secret);

    // Forcar um tipo de uma variavel com "as" - Hackzinho do TypeScript
    const { sub } = decoded as TokenPayload;

    // Fazer outro hackezinho pra sobrescreve tipos de uma biblioteca. @types/express.d.ts
    request.user = {
      id: sub,
    };

    // Se estiver com o jwt autenticado/correto, pode continuar usando a aplicacao.
    return next();
  } catch {
    // Se nao estiver autenticado
    throw new AppError('Invalid JWT token', 401);
  }
}

import { Router } from 'express';

import AuthenticateUserService from '../services/AuthenticateUserService';

const sessionsRouter = Router();

sessionsRouter.post('/', async (request, response) => {
  try {
    const { email, password } = request.body;

    // Criar um service para fazer as regras de negocio
    const authenticateUser = new AuthenticateUserService();

    const { user, token } = await authenticateUser.execute({
      email,
      password,
    });

    // Nao mostrar a senha no return de user.
    delete user.password;

    return response.json({ user, token });
  } catch (err) {
    return response.status(err.statusCode).json({ error: err.message });
  }
});

export default sessionsRouter;
