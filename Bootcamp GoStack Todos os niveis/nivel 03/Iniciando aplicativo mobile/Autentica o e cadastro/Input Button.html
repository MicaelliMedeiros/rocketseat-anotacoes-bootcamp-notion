Na src criar a pasta components e dentro a pasta Input e dentro os arquivos index.tsx e styles.ts

Instalar o React-Native-Vector-Icons

yarn add react-native-vector-icons

yarn add -D @types/react-native-vector-icons

No IOS
cd ios
pod install

Ir no ios/appgobarber/info.plist e adicionar o Feather

<string>RobotoSlab-Regular.ttf</string>
<string>Feather.ttf</string>

Ir no android/app/build.gradle - No final do arquivo

  iconFontNames: ['Feather.ttf']
]

apply from: "../../node_modules/react-native-vector-icons/fonts.gradle"

Input/index.tsx

import React from 'react';
import { TextInputProps } from 'react-native';

import { Container, TextInput, Icon } from './styles';

interface InputProps extends TextInputProps {
  name: string;
  icon: string;
}

const Input: React.FC<InputProps> = ({ name, icon, ...rest }) => {
  return (
    <Container>
      <Icon name={icon} size={20} color="#666360" />

      <TextInput
        keyboardAppearance="dark" // Somente IOS - o teclado fica dark
        placeholderTextColor="#666360"
        {...rest}
      />
    </Container>
  );
};

export default Input;

Button/index.tsx

import React from 'react';
import { RectButtonProperties } from 'react-native-gesture-handler';

import { Container, ButtonText } from './styles';

interface ButtonProps extends RectButtonProperties {
  children: string;
}

const Button: React.FC<ButtonProps> = ({ children, ...rest }) => (
  <Container {...rest}>
    <ButtonText>{children}</ButtonText>
  </Container>
);

export default Button;
